### Creating a basic Mint
```json
{
  "public_key": "YOUR_PUBLIC_KEY",
  "signature": "SIGNATURE",
  "payload": {
    "title": "Lambo",
    "fraction_count": 1000,
    "description": "Red Lambo Super Car",
    "owner_address": "YOUR_ADDRESS"
  }
}
```

### Creating a basic Mint with Metadata
```json
{
  "public_key": "YOUR_PUBLIC_KEY",
  "signature": "SIGNATURE",
  "payload": {
    "title": "Lambo",
    "fraction_count": 1000,
    "description": "Red Lambo Super Car",
    "metadata": {
      "vehicle": "car",
      "vin": "23123213213213",
      "wheels": 6
    },
    "owner_address": "YOUR_ADDRESS"
  }
}
```

### Creating a basic Mint with Invoice Signature requirements
```json
{
  "public_key": "YOUR_PUBLIC_KEY",
  "signature": "SIGNATURE",
  "payload": {
    "title": "Lambo",
    "fraction_count": 1000,
    "description": "Red Lambo Super Car",
    "owner_address": "YOUR_ADDRESS",
    "signature_requirement_type": "REQUIRES_MIN_SIGNATURES",
    "asset_managers": [
      {
        "name": "Asset Manager",
        "public_key": "AM_PUBLIC_KEY",
        "url": "https://example.com/assetManager"
      },
      {
        "name": "Asset Manager 2",
        "public_key": "AM_PUBLIC_KEY_2",
        "url": "https://example.com/assetManager2"
      }
    ],
    "min_signatures": 2
  }
}
```
